# clonare (se non già fatto)
git clone https://github.zhaw.ch/milanste/IoT.git
cd IoT

# virtualenv + dipendenze
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt

sudo systemctl enable --now mosquitto
